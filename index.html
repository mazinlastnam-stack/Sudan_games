// استبدل دالة checkout() في index.html بالدالة التالية: function checkout() { if (!currentUser) { showToast("الرجاء تسجيل الدخول أولاً", "error"); openAuthModal('login'); return; } const cart = currentUser.cart || []; if (cart.length === 0) { showToast("السلة فارغة، أضف منتجات أولاً", "error"); return; } const total = cart.reduce((sum, item) => sum + (item.price * item.quantity), 0); if (currentUser.walletBalance < total) { showToast(`رصيدك غير كافٍ. المطلوب: ${total.toLocaleString()} ج.س`, "error"); return; } // إنشاء طلب شراء جديد لكل عنصر في السلة const purchaseRequests = JSON.parse(localStorage.getItem('sudangames_purchase_requests') || '[]'); for (const item of cart) { purchaseRequests.push({ id: Date.now() + Math.random() * 10000, customerId: currentUser.id, customerName: currentUser.name, productId: item.id, productName: item.name, price: item.price, quantity: item.quantity, status: 'pending', createdAt: new Date().toISOString() }); } localStorage.setItem('sudangames_purchase_requests', JSON.stringify(purchaseRequests)); // تفريغ السلة currentUser.cart = []; saveUserState(); updateCartBadge(); closeCart(); showToast(`تم إرسال طلب الشراء للمراجعة. سيتم إعلامك بعد الموافقة.`, "success"); }
